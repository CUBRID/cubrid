CREATE TABLE "sup2"
CLASS ATTRIBUTE(
"float" smallint DEFAULT 1
)
(
"bigint" bigint,
"numeric1" numeric(15,0) AUTO_INCREMENT NOT NULL UNIQUE,
"numeric2" numeric(17,2) DEFAULT 12.50 NOT NULL,
"float" float DEFAULT 12 NOT NULL,
"date" date DEFAULT DATE'12/12/2009',
"smallint2" smallint,
"cache" sup1,

);
