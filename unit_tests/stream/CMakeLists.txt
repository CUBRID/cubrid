project (test_stream)

set (TEST_STREAM_SRC
  test_main.cpp
  test_stream.cpp
  )
set (TEST_STREAM_H
  test_stream.hpp
  )
SET_SOURCE_FILES_PROPERTIES(
  ${TEST_STREAM_SRC}
  PROPERTIES LANGUAGE CXX
  )

add_executable(test_stream
  ${TEST_STREAM_SRC}
  ${TEST_STREAM_H}
  )

target_compile_definitions(test_stream PRIVATE
  SERVER_MODE
  ${COMMON_DEFS}
  )

target_include_directories(test_stream PRIVATE
  ${TEST_INCLUDES}
  )

target_link_libraries(test_stream PRIVATE
  test_common
  )
if(UNIX)
  target_link_libraries(test_stream PRIVATE
    cubrid
    )
elseif(WIN32)
	target_link_libraries(test_stream PRIVATE
    cubrid-win-lib
    )
else()
  message( SEND_ERROR "Stream unit testing is for unix/windows")
endif ()
